    def start_listening
      @interface.listen_for_hid_interrupts { | interface |
        while(1)
          data = interface.next_interrupt
          @usages_to_watch.each_pair { |usage, command|
            usage_val = data.get_usage_value(usage)
            if(usage_val != command[0])
              command[0] = usage_val
              command[1].call(usage_val)
            end
          }
        end
      }
    end
